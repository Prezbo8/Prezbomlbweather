// ── LAZY FETCH PATCH (only load weather when team is opened) ─────────────────

async function fetchTeamData(abbr){

  if(allData[abbr] && !allData[abbr].err) return;

  const team=TEAMS.find(t=>t.abbr===abbr);

  if(!team) return;

  const url=`https://api.open-meteo.com/v1/forecast?latitude=${team.lat}&longitude=${team.lon}`

    +`&hourly=temperature_2m,relative_humidity_2m,surface_pressure,wind_speed_10m,`

    +`wind_direction_10m,wind_gusts_10m,apparent_temperature,precipitation_probability,precipitation`

    +`&timezone=auto&forecast_days=1&wind_speed_unit=mph&temperature_unit=fahrenheit`;

  try{

    const json=await fetch(url).then(r=>r.json());

    const h=json.hourly; const hours=[];

    for(let i=0;i<h.time.length;i++){

      const t=new Date(h.time[i]);

      if(t.getHours()<6) continue;

      hours.push({

        time:t, temp:Math.round(h.temperature_2m[i]), feels:Math.round(h.apparent_temperature[i]),

        humid:h.relative_humidity_2m[i], pressure:Math.round(h.surface_pressure[i]),

        wSpeed:Math.round(h.wind_speed_10m[i]), wDir:h.wind_direction_10m[i],

        wCard:cardinal(h.wind_direction_10m[i]), gusts:Math.round(h.wind_gusts_10m[i]),

        pPct:h.precipitation_probability[i], pIn:h.precipitation[i],

      });

    }

    allData[abbr]={abbr,hours,err:false};

  }catch{allData[abbr]={abbr,hours:[],err:true};}

}

function parseGameTime(str){if(!str)return null;str=str.trim().toUpperCase();let h,m=0,isPM=str.includes('PM'),isAM=str.includes('AM');str=str.replace(/[APM\s]/g,'');if(str.includes(':'))[h,m]=str.split(':').map(Number);else if(str.length<=2)h=Number(str);else{h=Number(str.slice(0,-2));m=Number(str.slice(-2));}if(isNaN(h))return null;if(isPM&&h<12)h+=12;if(isAM&&h===12)h=0;if(!isPM&&!isAM&&h<10)h+=12;return h*60+m;}

function minToLabel(mins){if(mins==null)return'—';let h=Math.floor(mins/60)%24,m=mins%60;const ap=h>=12?'PM':'AM';const d=h>12?h-12:(h===0?12:h);return`${d}:${String(m).padStart(2,'0')} ${ap}`;}

function getHourData(abbr,targetMins){const d=allData[abbr];if(!d||d.err||!d.hours.length)return null;const th=Math.floor(targetMins/60);return d.hours.find(h=>h.time.getHours()===th)||null;}

function renderTable(){

  const q=document.getElementById('search').value.toLowerCase().trim();

  const divF=document.getElementById('div-filter').value;

  const sortK=document.getElementById('sort').value;

  let data=TEAMS.filter(t=>{

    const mQ=!q||t.abbr.toLowerCase().includes(q)||t.stadium.toLowerCase().includes(q);

    const mD=divF==='all'||t.division===divF;

    return mQ&&mD;

  });

  if(sortK==='game') data.sort((a,b)=>{

    const ga=parseGameTime(gameTimes[a.abbr])??9999;

    const gb=parseGameTime(gameTimes[b.abbr])??9999;

    return ga-gb||a.abbr.localeCompare(b.abbr);

  }); else data.sort((a,b)=>a.abbr.localeCompare(b.abbr));


  document.getElementById('status-pill').innerHTML=`<b>${data.length}</b> stadiums`;

  if(!data.length){document.getElementById('tbody').innerHTML=`<tr class="empty-row"><td colspan="7">No stadiums match</td></tr>`;return;}


  document.getElementById('tbody').innerHTML=data.map((t,idx)=>{

    const d=allData[t.abbr];

    const now=d?.hours?.find(h=>h.time.getHours()===new Date().getHours())||d?.hours?.[0];

    const gt=gameTimes[t.abbr];

    const isOpen=openTeam===t.abbr;

    const badge=`<span class="team-badge" style="background:${t.color}">${t.abbr}</span>`;

    const tempStr=now?`<span class="summary-temp ${tCol(now.temp)}">${now.temp}°F</span>`:'<span style="color:var(--muted)">—</span>';

    const windStr=now?`<span style="font-weight:600" class="${wColRow(now.wSpeed)}">${now.wSpeed} mph ${now.wCard}</span>`:'—';

    const precipStr=now?`<span style="font-weight:600" class="${pColRow(now.pPct)}">${now.pPct}%</span>`:'—';

    const gtStr=gt?`<span class="has-game">⚾ ${gt.toUpperCase()}</span>`:`<span class="no-game">no game set</span>`;

    return `

    <tr class="team-row${isOpen?' open':''}" id="tr-${t.abbr}" style="animation-delay:${idx*.02}s" onclick="toggleTeam('${t.abbr}')">

      <td>${badge}</td>

      <td><div class="stadium-name">${t.stadium}</div><div class="stadium-div">${t.division}</div></td>

      <td>${gtStr}</td><td>${tempStr}</td><td>${windStr}</td><td>${precipStr}</td>

      <td><span class="expand-arrow">▼</span></td>

    </tr>

    <tr class="expand-row${isOpen?' open':''}" id="er-${t.abbr}">

      <td colspan="7">${isOpen?buildExpandPanel(t.abbr):''}</td>

    </tr>`;

  }).join('');

}

function setTime(abbr){

  const val=document.getElementById(`ti-${abbr}`)?.value?.trim();

  if(val) gameTimes[abbr]=val; else delete gameTimes[abbr];

  const cell=document.querySelector(`#er-${abbr} td`);

  if(cell) cell.innerHTML=buildExpandPanel(abbr);

  updateRowGT(abbr);

}

function clearTime(abbr){

  delete gameTimes[abbr];

  const cell=document.querySelector(`#er-${abbr} td`);

  if(cell) cell.innerHTML=buildExpandPanel(abbr);

  updateRowGT(abbr);

}

function liveUpdate(abbr){

  clearTimeout(liveUpdate._t);

  liveUpdate._t=setTimeout(()=>{

    const val=document.getElementById(`ti-${abbr}`)?.value?.trim();

    if(val) gameTimes[abbr]=val;

    const win=document.getElementById(`windows-${abbr}`);

    if(!win) return;

    const parsed=parseGameTime(val);

    if(parsed!=null) win.innerHTML=`<div class="windows-grid">

      ${buildWindowCard(abbr,-60,'1H Before',false)}

      ${buildWindowCard(abbr,0,'First Pitch',true)}

      ${buildWindowCard(abbr,60,'+1 Hour',false)}

      ${buildWindowCard(abbr,120,'+2 Hours',false)}

      ${buildWindowCard(abbr,180,'+3 Hours',false)}

    </div>`;

    else win.innerHTML=`<div class="no-time-prompt">Enter a game time above to see the <span>5-window forecast</span></div>`;

  },350);

}

}
